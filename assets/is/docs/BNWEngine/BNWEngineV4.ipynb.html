worker_processes = [1, 2, 3, 4, 5, 6, 7, 8]
time_BNWEngineV4 = [7.165, 6.739, 6.208, 6.001, 5.176, 5.153, 4.405, 3.940]
time_BNWEngineV2 = [14.60 for _ in range(len(time_BNWEngineV4))]
absolute_difference = [100 * (v2 - v4) / ((v2 + v4) / 2)  for v2, v4 in zip(time_BNWEngineV2, time_BNWEngineV4)]

# Create subplots (2 rows, 1 column)
fig, axs = plt.subplots(1, 2, figsize=(10, 5))

# First subplot: Execution times for BNWEngineV4 and BNWEngineV2
axs[0].plot(worker_processes, time_BNWEngineV4, label='BNWEngineV4', marker='o', color='blue')
axs[0].plot(worker_processes, time_BNWEngineV2, label='BNWEngineV2', marker='o', color='red')

# Labels and title for the first plot
axs[0].set_xlabel('Number of Worker Processes')
axs[0].set_ylabel('Time Needed (seconds)')
axs[0].set_title('Time Needed Against Number of Workers')
axs[0].grid(True)
axs[0].legend()

# Second subplot: Absolute difference between BNWEngineV4 and BNWEngineV2
axs[1].plot(worker_processes, absolute_difference, marker='o', color='green', label='Percentage Difference')

# Labels and title for the second plot
axs[1].set_xlabel('Number of Worker Processes')
axs[1].set_ylabel('Percentage Difference (%)')
axs[1].set_yticks(np.linspace(0, 200, 11))
axs[1].set_title('Percentage Difference Against Number of Worker')
axs[1].grid(True)
axs[1].legend()

# Adjust layout to prevent overlapping
plt.tight_layout()
